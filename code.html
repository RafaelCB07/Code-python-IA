IA analyse de sentiment

L'IA d'analyse de sentiment détecte et évalue les émotions dans vos phrases en identifiant des mots et des adverbes positifs ou négatifs, fournissant ainsi un score de sentiment précis. C'est peut être l'occasion pour vous d'exploiter cette base pour adresser une fonction plus précise de cette IA sur des critiques de film par exemple !

from transformers import pipeline

                # Créer un pipeline d'analyse de sentiment
                sentiment_pipeline = pipeline("sentiment-analysis")
                
                # Mots et leurs poids associés
                positive_words = {
                    "excellent": 0.5,
                    "super": 0.4,
                    "génial": 0.5,
                    "formidable": 0.5,
                    "heureux": 0.4,
                    "magnifique": 0.5,
                    "merveilleux": 0.5,
                    "aimé": 0.3,
                    "top": 0.4,
                    "plaisant": 0.3,
                }
                
                negative_words = {
                    "terrible": -0.5,
                    "horrible": -0.6,
                    "triste": -0.4,
                    "déprimant": -0.5,
                    "nul": -0.3,
                    "mauvais": -0.4,
                    "décevant": -0.4,
                    "en colère": -0.5,
                }
                
                # Adverbes qui modifient les sentiments
                adverb_weights = {
                    "très": 0.3,
                    "extrêmement": 0.4,
                    "un peu": 0.2,
                    "vraiment": 0.3,
                    "rarement": -0.2,
                    "jamais": -0.4,
                }
                
                # Mots de négation
                negation_words = ["pas", "aucun", "jamais", "sans", "nulle", "ne"]
                
                def analyze_sentiment(sentence):
                    """Analyse le sentiment d'une phrase et renvoie le type de sentiment."""
                    result = sentiment_pipeline(sentence)[0]
                    sentiment = result['label']
                    score = result['score']
                
                    # Ajustement du score en fonction des mots de négation et des mots-clés
                    words = sentence.lower().split()
                    adjusted_score = score
                
                    # Gérer les mots de négation
                    negation_found = any(word in negation_words for word in words)
                
                    # Ajuster le score pour chaque mot dans la phrase
                    for word in words:
                        # Mots positifs
                        if word in positive_words:
                            adjusted_score += positive_words[word]
                        # Mots négatifs
                        elif word in negative_words:
                            adjusted_score += negative_words[word]
                        # Adverbes
                        elif word in adverb_weights:
                            adjusted_score += adverb_weights[word]
                
                    # Appliquer l'effet de la négation
                    if negation_found:
                        adjusted_score *= -1  # Inverser le score si une négation est trouvée
                
                    # Évaluer le sentiment ajusté
                    if adjusted_score > 0.2:
                        adjusted_sentiment = "Très Positif"
                    elif adjusted_score > 0:
                        adjusted_sentiment = "Légèrement Positif"
                    elif adjusted_score < -0.2:
                        adjusted_sentiment = "Très Négatif"
                    else:
                        adjusted_sentiment = "Légèrement Négatif"
                
                    return adjusted_sentiment, max(0, min(1, adjusted_score))  # Garder le score entre 0 et 1
                
                def main():
                    """Fonction principale pour exécuter l'IA d'analyse de sentiment."""
                    print("Bienvenue dans l'IA d'analyse de sentiment !")
                    while True:
                        user_input = input("Veuillez entrer une phrase (ou tapez 'exit' pour quitter) : ")
                        if user_input.lower() == 'exit':
                            print("Merci d'avoir utilisé l'IA d'analyse de sentiment !")
                            break
                        
                        sentiment, score = analyze_sentiment(user_input)
                        print(f"Le sentiment de la phrase est : {sentiment} (score : {score:.2f})\n")
                
                if __name__ == "__main__":
                    main()

IA de gestion du temps

L'Assistant de Gestion du Temps est conçu pour aider les utilisateurs à organiser leurs tâches et à maximiser leur productivité. Grâce à une interface intuitive, cette IA permet d'ajouter, de supprimer et de gérer des tâches, tout en offrant des conseils pratiques pour améliorer l'efficacité. 


                import time
                import re
                from datetime import datetime, timedelta
                from collections import defaultdict
                
                class Task:
                    def __init__(self, title, due_date=None, priority=1, status='to do'):
                        self.title = title
                        self.due_date = due_date
                        self.priority = priority
                        self.status = status
                
                    def __str__(self):
                        return f"{self.title} (due: {self.due_date.strftime('%Y-%m-%d %H:%M') if self.due_date else 'No due date'}, priority: {self.priority}, status: {self.status})"
                
                class TimeManagementAssistant:
                    def __init__(self):
                        self.tasks = []
                        self.task_stats = defaultdict(list)  # Store time spent on tasks
                
                    def add_task(self, title, due_date=None, priority=1):
                        task = Task(title, due_date, priority)
                        self.tasks.append(task)
                        print(f"Tâche ajoutée : {task}")
                
                    def remove_task(self, title):
                        self.tasks = [task for task in self.tasks if task.title != title]
                        print(f"Tâche '{title}' supprimée.")
                
                    def list_tasks(self):
                        if not self.tasks:
                            print("Aucune tâche à afficher.")
                        else:
                            print("Liste des tâches :")
                            sorted_tasks = sorted(self.tasks, key=lambda x: (x.priority, x.due_date))
                            for task in sorted_tasks:
                                print(f"- {task}")
                
                    def parse_due_date(self, date_str):
                        """Essaie de parser différentes formats de date."""
                        formats = [
                            "%Y-%m-%d %H:%M",
                            "%d/%m/%Y %H:%M",
                            "%d-%m-%Y %H:%M",
                            "%Y/%m/%d %H:%M",
                            "%Y-%m-%d",
                            "%d/%m/%Y",
                            "%d-%m-%Y",
                            "%Y/%m/%d",
                            "%H:%M %d/%m/%Y",
                            "%H:%M %Y-%m-%d",
                        ]
                        for fmt in formats:
                            try:
                                return datetime.strptime(date_str, fmt)
                            except ValueError:
                                continue
                        return None
                
                    def set_reminder(self, task_title, reminder_time):
                        while True:
                            now = datetime.now()
                            if now >= reminder_time:
                                print(f"Rappel : Il est temps de {task_title}!")
                                break
                            time.sleep(30)
                
                    def productivity_tips(self):
                        tips = [
                            "Essayez la technique Pomodoro : travaillez pendant 25 minutes, puis faites une pause de 5 minutes.",
                            "Priorisez vos tâches : commencez par les tâches les plus importantes.",
                            "Utilisez des applications de gestion du temps pour rester organisé.",
                            "Ne laissez pas les tâches s'accumuler : faites un peu chaque jour!",
                            "Essayez de limiter vos tâches à 3-5 tâches principales par jour pour ne pas vous sentir accablé."
                        ]
                        print("Conseils de productivité :")
                        for tip in tips:
                            print(f"- {tip}")
                
                    def suggest_task_based_on_time(self):
                        upcoming_tasks = sorted(
                            [task for task in self.tasks if task.due_date],
                            key=lambda x: x.due_date
                        )
                        if not upcoming_tasks:
                            print("Aucune tâche à échéance imminente.")
                            return
                        
                        print("Tâches à venir :")
                        for task in upcoming_tasks:
                            remaining_time = task.due_date - datetime.now()
                            if remaining_time <= timedelta(hours=1):
                                print(f"Suggestion : Vous devriez vous concentrer sur '{task.title}', il reste moins d'une heure avant l'échéance.")
                            elif remaining_time <= timedelta(days=1):
                                print(f"Suggestion : Pensez à terminer '{task.title}' bientôt, l'échéance est demain.")
                
                    def analyze_productivity(self):
                        if not self.tasks:
                            print("Aucune tâche pour analyser la productivité.")
                            return
                
                        completed_tasks = [task for task in self.tasks if task.status == 'completed']
                        if completed_tasks:
                            print(f"Vous avez complété {len(completed_tasks)} tâche(s) depuis votre dernière session.")
                            for task in completed_tasks:
                                print(f"- {task.title}")
                        else:
                            print("Aucune tâche n'a encore été complétée.")
                
                    def update_task_status(self, title, status):
                        for task in self.tasks:
                            if task.title == title:
                                task.status = status
                                print(f"Tâche '{title}' mise à jour avec le statut '{status}'.")
                                return
                        print(f"Aucune tâche trouvée avec le titre '{title}'.")
                
                    def generate_report(self):
                        total_tasks = len(self.tasks)
                        completed_tasks = [task for task in self.tasks if task.status == 'completed']
                        print("\n--- Rapport de Temps ---")
                        print(f"Tâches totales : {total_tasks}")
                        print(f"Tâches complétées : {len(completed_tasks)}")
                        for task in completed_tasks:
                            print(f"- {task.title} (due: {task.due_date})")
                        print("-----------------------")
                
                    def start_timer(self, duration, pause_duration):
                        print(f"Début de la session de révision pour {duration} minutes !")
                        time.sleep(duration * 60)  # Convertit les minutes en secondes
                        print("Temps écoulé ! Prenez une pause de {pause_duration} minutes.")
                        time.sleep(pause_duration * 60)  # Pause
                        print("Pause terminée ! Prêt pour la prochaine session.")
